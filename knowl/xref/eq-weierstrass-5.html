\begin{equation} \begin{aligned} \sum_{k \in C_x}p_{n,k}(x) &\leq \sum_{k \in C_x}p_{n,k}(x) \frac{(k-nx)^2}{\delta^2n^2}\\ &\leq \sum_{k=0}^n p_{n,k}(x) \frac{(k-nx)^2}{\delta^2n^2} \\ &= \frac{1}{n^2\delta^2} \sum_{k=0}^n(k-nx)^2p_{n,k}(x) \\ &= \frac{x(1-x)}{n\delta^2}\\ &\leq \frac{1}{4n\delta^2}, \end{aligned}\tag{1.5} \end{equation}
